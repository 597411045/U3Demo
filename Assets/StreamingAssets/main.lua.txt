function Lua_OnDamage(obj)
    obj.hp = obj.hp - 1;
    obj.obj[obj.hp]:SetActive(false);

    obj.hp = obj.hp - 1;

    if (obj.hp > 0) then
        obj.obj[obj.hp]:SetActive(false);
    end

    if (obj.hp <= 0) then

        CS.FacadeGame.GetInstance(null):SendNotification(CS.GameConst.MSG_GAME_LOSE);
    end
end

}

function Enemy.Lua_Enemy_Update(obj)

    if obj.state == CS.eState.Move then
        obj.transform:Translate(CS.UnityEngine.Vector3.right * obj.speed * CS.UnityEngine.Time.deltaTime);
        if obj.transform.position.x >= CS.UnityEngine.Camera.main.orthographicSize * CS.GameConst.Screen_Ratio then
            obj.transform.position = CS.UnityEngine.Vector3(-CS.UnityEngine.Camera.main.orthographicSize * CS.GameConst.Screen_Ratio, obj.transform.position.y);
        end
    end

end
