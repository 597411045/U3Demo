local Tree = {}
local a = 10;
Tree.Color = "Red";
Tree.Height = 10;

function Tree:new(color, h)
    local obj = {}
    obj.Color = color or Tree.Color;
    obj.Height = h or Tree.Height;
    setmetatable(obj, {
        __index = Tree
    })
    return obj;
end

function Tree:grow(int)
    self.Height = self.Height + int + a;
end

return Tree;
